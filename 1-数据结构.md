ll exgcd(ll a, ll b, ll &x, ll &y){
    if(!b){
        x = 1, y = 0;
        return a;
    }
    ll d = exgcd(b, b % a, x, y);
    ll t = x;
    x = y;
    y = t - a / b * y;
    return d;
}
void Solve(){
    ll a1, b1, a2, b2, L, R;
    cin >> a1 >> b1 >> a2 >> b2 >> L >> R;
    ll A = a1, B = -a2, C = b2 - b1;
    //A * k1 + B * k2 = C
    // A * x + B * y = gcd(A, B);
    ll x, y;
    ll g = exgcd(A, B, x, y);
    if(C % g){
        cout << -1 << endl;
        return;
    }  
    // A * x * C / g + B * x * C / g = C
    // k1 = (x * C - a2 * n) / g;
    // k2 = (y * C - a1 * n) / g;
    ll f = 1;
    if(g < 0){
        g = -g;
        x = -x;
        y = -y;
        f = -1;
    }
    ll mi = -1e18, mx = 1e18;
    ll l = -1e9, r = 1e9;
    while(l < r){
        ll mid = l + r >> 1;
        if(L <= (x * C - f * a2 * mid) / g * a1 + b1) r = mid;
        else l = mid + 1;
    }
    mi = max(mi, r);
    l = -1e9, r = 1e9;
    while(l < r){
        ll mid = l + r + 1 >> 1;
        if(R >= (x * C - f * a2 * mid) / g * a1 + b1) l = mid;
        else r = mid - 1;
    }
    mx = min(mx, r);
    l = -1e9, r = 1e9;
    while(l < r){
        ll mid = l + r >> 1;
        if(L <= (y * C - f * a1 * mid) / g * a2 + b2) r = mid;
        else l = mid + 1;
    }
    mi = max(mi, r);
    l = -1e9, r = 1e9;
    while(l < r){
        ll mid = l + r + 1 >> 1;
        if(R >= (y * C - f * a1 * mid) / g * a2 + b2) l = mid;
        else r = mid - 1;
    }
    mx = min(mx, r);
    l = -1e9, r = 1e9;
    while(l < r){
        ll mid = l + r >> 1;
        if(0 <= (y * C - f * a1 * mid) / g) r = mid;
        else l = mid + 1;
    }

    l = -1e9, r = 1e9;

        ll mid = l + r >> 1;
        if(0 <= (x * C - f * a2 * mid) / g) r = mid;
        else l = mid + 1;
    }

    return;
}
